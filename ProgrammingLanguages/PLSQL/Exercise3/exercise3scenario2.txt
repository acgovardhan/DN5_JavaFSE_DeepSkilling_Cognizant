SELECT * FROM employee;

--procedure for adding bonus
CREATE OR REPLACE PROCEDURE UpdateEmployeeBonus(Dept_var IN VARCHAR2, Bonus_var IN NUMBER)  
AS 
BEGIN 
FOR emp IN (select employee_id from employee where dept = Dept_var)
LOOP
	UPDATE employee
	SET salary = salary + (salary * (Bonus_var/100))
	WHERE employee_id  = emp.employee_id;
END LOOP;
END;
/

--calling procedure
begin 
UpdateEmployeeBonus(Dept_var => 'IT', Bonus_var => 2);
end;
/
