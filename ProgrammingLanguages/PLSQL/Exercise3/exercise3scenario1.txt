    customer_id VARCHAR2(10),
    customer_name VARCHAR2(50),
    accountType VARCHAR(2),
    balance NUMBER

--procedure for processing interest
CREATE OR REPLACE PROCEDURE ProcessMonthlyInterest
AS 
BEGIN 
FOR acc IN (select customer_id from accounts where accountType = 'SA')
LOOP
	UPDATE accounts
	SET balance = balance + (balance * 0.01)
	WHERE customer_id  = acc.customer_id;
END LOOP;
END;
/

--calling procedure
begin 
ProcessMonthlyInterest;
end;
/
